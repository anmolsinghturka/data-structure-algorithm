#include<iostream>
using namespace std;

typedef struct BSTNode{
int data;
BSTNode *left;
BSTNode *right;
};

BSTNode* Find(BSTNode*root,int data)
{
    if(root==NULL)
        return(NULL);
    if(data<root->data)
        return(root->left);
    else if(data>root->right)
        return(root->right);
    return(root);

}

void Insert(BSTNode *root,int data)
{
    BSTNode *par;
    BSTNode *node=new BSTNode();

    node->data=data;
    node->left=NULL;
    node->right=NULL;

    if(root==NULL)
        root=node;
    else
    {
        par=root;
    while(par!=NULL)
    {
        if(par->data>data)
        {
            if(par->left==NULL)
                par->left=node;
            par=par->left;
        }
        else if(par->data<data)
        {
            if(par->right==NULL)
                par->right=node;
            par=par->right;
        }
    }

    }

}

void preorder(BSTNode *root)
{
	if(root)
	{
		cout<<root->info<<" ";
		preorder(root->left);
		preorder(root->right);
	}
}


void Inorder(BSTNode *root)
{
	if(root)
	{
		preorder(root->left);
		cout<<root->info<<" ";
		preorder(root->right);
	}
}


void postorder(BSTNode *root)
{
	if(root)
	{
		preorder(root->left);
		preorder(root->right);
		cout<<root->info<<" ";
	}
}
